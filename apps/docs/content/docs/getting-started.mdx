## 1. Install

```bash
npm install @sargonpiraev/seokit @playwright/test
```

## 2. Extend

Create `src/test/seokit.ts`:

```typescript
import { expect as baseExpect, test } from "@playwright/test";
import { extendSeokitExpect } from "@sargonpiraev/seokit";

const expect = extendSeokitExpect(baseExpect);

export { test, expect };
```

Import `test` and `expect` from this file in every SEO spec — not from `@playwright/test` directly.

## 3. Playwright project

In `playwright.config.ts`, keep SEO specs in their own project:

```typescript
import { defineConfig } from "@playwright/test";

export default defineConfig({
  testDir: "./src",
  projects: [
    { name: "app", testMatch: "**/*.test.ts" },
    { name: "seokit", testMatch: "**/*.seokit.spec.ts" },
  ],
  use: {
    baseURL: process.env.PLAYWRIGHT_BASE_URL ?? "http://localhost:3000",
  },
});
```

```bash
npx playwright test --project seokit
```

## 4. First colocated spec

Place the spec next to the page, for example:

`src/app/[locale]/products/[id]/page.seokit.spec.ts`

```typescript
import { routing } from "@/i18n/routing";
import { test, expect } from "@/test/seokit";

const siteUrl = process.env.NEXT_PUBLIC_SITE_URL;
const id = "alpha";

for (const locale of routing.locales) {
  const pageUrl = new URL(`/${locale}/products/${id}`, siteUrl).toString();

  test(pageUrl, async ({ page }) => {
    await page.goto(pageUrl);

    await expect(page).toHaveHtmlLang(locale);
    await expect(page).toHaveCanonical(pageUrl);
    await expect(page).toHaveSelfAlternate(locale, pageUrl);

    const xDefaultUrl = new URL(`/${routing.defaultLocale}/products/${id}`, siteUrl).toString();
    await expect(page).toHaveXDefaultAlternate(xDefaultUrl);

    for (const alternateLocale of routing.locales) {
      const alternateUrl = new URL(`/${alternateLocale}/products/${id}`, siteUrl).toString();
      await expect(page).toHaveAlternate(alternateLocale, alternateUrl);
    }
  });
}
```

## 5. How it works

The example above does three things:

1. **`routing.locales`** — iterate the same next-intl locale list your app already uses.
2. **`id`** — dynamic segments are just values you pass in the spec (`"1"`).
3. **Matchers** — assert against absolute URLs built with `new URL(..., siteUrl)`. The same `pageUrl` is the test title and what you pass to `page.goto`.

See all matchers in [Matchers](/docs/matchers).

Working example: [`apps/example`](https://github.com/sargonpiraev/seokit/tree/main/apps/example).
